int soma_saldo(int saldo, int valor){


    saldo = saldo + valor;
    return saldo;


}

int subtrai_saldo(int saldo, int valor){


    saldo = saldo - valor;
    return saldo;


}

int saldo_disponivel(int salario,int divida){
    int saldo = 0;
    saldo = soma_saldo(saldo,salario);
    saldo = subtrai_saldo(saldo,divida);
    return saldo;
}
